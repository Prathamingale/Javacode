Code:method and constructor overloading
public class College {
	String collegeName;
	int estYear;
	
	College()
	{
		System.out.println("inside defualt constructor");
		collegeName="mmcoe";
		estYear=2008;
		System.out.println("college name is :" +collegeName);
		System.out.println("establishment year is :"+estYear);
	}
	
	College(String Name , int year)
	{
	
		System.out.println("inside parameterized constructor:");
		collegeName = Name;
		estYear = year;
		System.out.println("college name is :" +collegeName);
		System.out.println("establishment year is :"+estYear);
	}
	public static void main(String[] args) {
	
		College c1 = new College();
		College c2 = new College("mmcc",1980);
	}
}
