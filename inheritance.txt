class Player
{
	public String Name;
	public int birthYear;
	public float weight,height;

	Player()
	{
		Name=null;
		weight=0.0f;
		height=0.0f;
		birthYear=1900;
	}

	Player(String n , int by , float w , float h)
	{
		Name=n;
		birthYear=by;
		weight=w;
		height=h;

	}
}

class cricket extends Player
{
	int No_Matches;
	int centuries;
	int runs;
	int wickets;

	cricket(String n , int by , float w , float h , int NoM , int c , int r , int wk )
	{
		super(n,by,w,h);
		No_Matches=NoM;
		centuries=c;
		runs=r;
		wickets=wk;
	}

	void display()
	{
		System.out.println("Name : "+Name);
		System.out.println("birth Year:" +birthYear);
		System.out.println("weight:"+weight);
		System.out.println("height:"+height);
		System.out.println("no of matches:"+No_Matches);
		System.out.println("centuries:"+centuries);
		System.out.println("runs:"+runs);
		System.out.println("wickets:"+wickets);
	}
}

class football extends Player

	int No_Matches;

	football(String n , int by , float w , float h , int NoM , int g)

}

		cricket c = new cricket("Rohit Sharma" , 1987 , 70.0f, 5.9f , 255 , 55 , 10000 , 70);
