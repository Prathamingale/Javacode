import java.util.Scanner;


public class Rectangle {
	Scanner sc = new Scanner(System.in);
	float length,width,area;
	String colour;
	
	void getData()
	{
		System.out.println("enter length and width of rectangle");
		length=sc.nextFloat();
		width=sc.nextFloat();
		System.out.println("enter colour for rectangle");
		colour=sc.next();
	}
	
	void calculate()
	{
		area=length*width;
		System.out.println("area of rectangle is:"+area);
	}
	
	public static void main(String[] args) {
	
		Rectangle r1 = new Rectangle();
		Rectangle r2 = new Rectangle();
		
		System.out.println("enter values for Rectangle 1");
		r1.getData();
		r1.calculate();
		
		System.out.println("enter values for Rectangle 2");
		r2.getData();
		r2.calculate();
		
		if((r1.area==r2.area)&&(r1.colour.equals(r2.colour)))
		{
			System.out.println("rectangle matching");
		}
		else {
			System.out.println("not matching");
		}
	}

}
